    <script>

        function toggleDarkMode() {

            const html = document.documentElement;

            const icon = document.getElementById('theme-icon');

            if (html.classList.contains('dark')) {

                html.classList.remove('dark');

                icon.innerText = '🌙';

            } else {

                html.classList.add('dark');

                icon.innerText = '☀️';

            }

        }


        function calculate() {

            const signInInput = document.getElementById('signIn').value;

            const leaveHours = parseFloat(document.getElementById('leaveHours').value);

            const resultBox = document.getElementById('resultBox');

            

            if (!signInInput) return;


            // 1. 基本下班時間計算 (含1小時午休)

            const [h, m] = signInInput.split(':').map(Number);

            let signDate = new Date();

            signDate.setHours(h, m, 0);


            // 應出勤 8 小時 + 1 小時午休 = 9 小時

            let offWorkDate = new Date(signDate.getTime() + 9 * 60 * 60 * 1000);

            

            const offH = offWorkDate.getHours().toString().padStart(2, '0');

            const offM = offWorkDate.getMinutes().toString().padStart(2, '0');

            document.getElementById('offWorkTime').innerText = `${offH}:${offM}`;


            if (leaveHours <= 0) {

                resultBox.classList.add('hidden');

                return;

            }


            resultBox.classList.remove('hidden');


            // 2. 最早簽退時間 = 預計下班時間 - 請假時數

            let exitDate = new Date(offWorkDate.getTime() - leaveHours * 60 * 60 * 1000);

            const exitH = exitDate.getHours().toString().padStart(2, '0');

            const exitM = exitDate.getMinutes().toString().padStart(2, '0');

            document.getElementById('earliestExit').innerText = `${exitH}:${exitM} 即可簽退`;


            // 3. 系統假單建議 (針對下班前請假情境)

            // 邏輯：系統上限 17:30，往前推請假時數

            let sysEnd = new Date();

            sysEnd.setHours(17, 30, 0);

            let sysStart = new Date(sysEnd.getTime() - leaveHours * 60 * 60 * 1000);


            const sSH = sysStart.getHours().toString().padStart(2, '0');

            const sSM = sysStart.getMinutes().toString().padStart(2, '0');

            const sEH = sysEnd.getHours().toString().padStart(2, '0');

            const sEM = sysEnd.getMinutes().toString().padStart(2, '0');

            

            document.getElementById('suggestedLeave').innerText = `${sSH}:${sSM} ~ ${sEH}:${sEM}`;

        }


        // 初始化計算一次

        window.onload = calculate;
